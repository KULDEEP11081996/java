import java.util.Scanner;

class SumOfDigits {
    public static void main(String[] args) {
        int num;
        int sum;
        Scanner k = new Scanner(System.in);
        System.out.print("Enter a number: ");
        num = k.nextInt();

        sum = 0;

        while (num != 0) {
            int digit = num % 10;
            sum += digit;
            num /= 10;
        }

        System.out.println("Sum of digits: " + sum);
        
    }
}
