import java.util.Scanner;

class shravan {
    public static void main(String[] args) {
        int age, weight;
        System.out.println("Enter the age:");
        Scanner g = new Scanner(System.in);
        age = g.nextInt();
        System.out.println("Enter the weight:");
        weight = g.nextInt();
        
        if (age >= 18) {
            if (weight >= 50) {
                System.out.println("Eligible for blood donation.");
            } else {
                System.out.println("Not eligible for blood donation. Weight must be at least 50.");
            }
        } else {
            System.out.println("Not eligible for blood donation. Age must be at least 18.");
        }
    }
}
